```sql
CREATE VIEW rekap_agama_jenis_kelamin AS (
 SELECT pegawai."JENIS_KELAMIN",
    agama."ID",
    agama."NAMA",
    count(*) AS total
   FROM (kepegawaian.pegawai
     LEFT JOIN kepegawaian.agama ON ((pegawai."AGAMA_ID" = agama."ID")))
  GROUP BY pegawai."JENIS_KELAMIN", agama."ID", agama."NAMA"
  ORDER BY agama."NAMA"
)
```
